```go
type VMUserSyncService interface {
    // 创建 VMuser
    CreateVMUser(ctx context.Context, tenant *Tenant) error
    
    // 更新 VMuser 配置
    UpdateVMUser(ctx context.Context, tenant *Tenant) error
    
    // 删除 VMuser
    DeleteVMUser(ctx context.Context, vmuserID string) error
    
    // 同步配额
    SyncQuota(ctx context.Context, tenant *Tenant) error
}

type VMUserSyncServiceImpl struct {
    vmOperator VMOperatorClient
    vmauth     *VMAuthClient
}

func (s *VMUserSyncServiceImpl) CreateVMUser(ctx context.Context, tenant *Tenant) error {
    // 1. 生成 API Key
    apiKey := generateAPIKey()
    
    // 2. 更新 vmauth 配置 (添加用户)
    user := VMAuthUser{
        Name:        tenant.VMUserID,
        Password:    apiKey,
        URLPrefixes: []string{
            "/insert/" + tenant.VMUserID,
            "/select/" + tenant.VMUserID,
        },
    }
    
    if err := s.vmauth.AddUser(ctx, user); err != nil {
        return err
    }
    
    // 3. 保存 API Key 到数据库
    tenant.VMUserKey = apiKey
    s.tenantRepo.Update(ctx, tenant)
    
    return nil
}

// 获取用户访问端点
func (s *VMUserSyncServiceImpl) GetAccessEndpoint(vmuserID string) string {
    return fmt.Sprintf("http://vmauth.platform:8421/insert/%s", vmuserID)
}
```

### 3.2 N9E 同步

```go
type N9ESyncService interface {
    // 同步租户 (创建团队)
    SyncTenant(ctx context.Context, tenant *Tenant) error
    
    // 同步用户
    SyncUser(ctx context.Context, user *User) error
    
    // 同步数据源
    SyncDatasource(ctx context.Context, tenant *Tenant, instances []Instance) error
    
    // 同步告警规则
    SyncAlertRules(ctx context.Context, tenant *Tenant, rules []AlertRule) error
}

type N9ESyncServiceImpl struct {
    n9eClient *N9EClient
}

func (s *N9ESyncServiceImpl) SyncTenant(ctx context.Context, tenant *Tenant) error {
    // 创建 N9E 团队 (对应租户)
    team := &N9ETeam{
        Name: tenant.VMUserID,
        Note: fmt.Sprintf("租户: %s, 部门: %s", tenant.TenantName, tenant.DeptName),
    }
    
    teamID, err := s.n9eClient.CreateTeam(ctx, team)
    if err != nil {
        return err
    }
    
    // 保存 team ID
    tenant.N9ETeamID = teamID
    s.tenantRepo.Update(ctx, tenant)
    
    return nil
}

func (s *N9ESyncServiceImpl) SyncUser(ctx context.Context, user *User) error {
    // 获取用户的租户
    tenant, err := s.tenantRepo.GetByID(ctx, user.TenantID)
    if err != nil {
        return err
    }
    
    // 创建 N9E 用户
    n9eUser := &N9EUser{
        Ident:    user.ID.String(),
        Nick:     user.Username,
        Password: user.PasswordHash,
        Ntype:    0, // 普通用户
        TeamID:   tenant.N9ETeamID,
    }
    
    // 租户管理员设置为团队管理员
    if user.Role == "admin" {
        n9eUser.IsAdmin = true
    }
    
    return s.n9eClient.CreateUser(ctx, n9eUser)
}

func (s *N9ESyncServiceImpl) SyncDatasource(ctx context.Context, tenant *Tenant, instances []Instance) error {
    // 为租户创建数据源
    for _, inst := range instances {
        var dsType, dsURL string
        
        switch inst.InstanceType {
        case "metrics":
            dsType = "prometheus"
            dsURL = fmt.Sprintf("http://vm-%s-vmselect:8481", tenant.VMUserID)
        case "logs":
            dsType = "loki"
            dsURL = fmt.Sprintf("http://vl-%s-victoria-logs-server:9428", tenant.VMUserID)
        }
        
        datasource := &N9EDatasource{
            Name:   inst.InstanceName,
            Type:   dsType,
            URL:    dsURL,
            Tenant: tenant.VMUserID,
        }
        
        if err := s.n9eClient.CreateDatasource(ctx, tenant.VMUserID, datasource); err != nil {
            log.Error("创建 N9E 数据源失败", "instance", inst.InstanceName, "error", err)
        }
    }
    
    return nil
}
```

### 3.3 Grafana 同步

```go
type GrafanaSyncService interface {
    // 同步租户 (创建组织)
    SyncTenant(ctx context.Context, tenant *Tenant) error
    
    // 同步用户
    SyncUser(ctx context.Context, user *User) error
    
    // 同步数据源
    SyncDatasource(ctx context.Context, tenant *Tenant, instances []Instance) error
    
    // 同步 Dashboard
    SyncDashboards(ctx context.Context, tenant *Tenant) error
}

type GrafanaSyncServiceImpl struct {
    grafanaClient *GrafanaClient
}

func (s *GrafanaSyncServiceImpl) SyncTenant(ctx context.Context, tenant *Tenant) error {
    // 创建 Grafana 组织 (对应租户)
    org := &GrafanaOrg{
        Name: tenant.VMUserID,
    }
    
    orgID, err := s.grafanaClient.CreateOrg(ctx, org)
    if err != nil {
        return err
    }
    
    // 保存 org ID
    tenant.GrafanaOrgID = orgID
    s.tenantRepo.Update(ctx, tenant)
    
    return nil
}

func (s *GrafanaSyncServiceImpl) SyncUser(ctx context.Context, user *User) error {
    tenant, err := s.tenantRepo.GetByID(ctx, user.TenantID)
    if err != nil {
        return err
    }
    
    // 租户管理员设置为组织管理员
    if user.Role == "admin" {
        return s.grafanaClient.SetOrgAdmin(ctx, tenant.VMUserID, user.ID.String())
    }
    
    // 普通用户添加为成员
    orgUser := &GrafanaOrgUser{
        Login: user.ID.String(),
        Role:  "Editor", // 或 "Viewer"
    }
    
    return s.grafanaClient.AddOrgUser(ctx, tenant.VMUserID, orgUser)
}

func (s *GrafanaSyncServiceImpl) SyncDatasource(ctx context.Context, tenant *Tenant, instances []Instance) error {
    // 为租户组织创建数据源
    for _, inst := range instances {
        var dsType, dsURL string
        
        switch inst.InstanceType {
        case "metrics":
            dsType = "prometheus"
            dsURL = fmt.Sprintf("http://vm-%s-victoria-metrics:8428", tenant.VMUserID)
            // 独享集群版
            if tenant.TemplateType == "dedicated_cluster" {
                dsURL = fmt.Sprintf("http://vm-%s-vmselect:8481", tenant.VMUserID)
            }
        case "logs":
            dsType = "loki"
            dsURL = fmt.Sprintf("http://vl-%s-victoria-logs-server:9428", tenant.VMUserID)
        }
        
        datasource := &GrafanaDatasource{
            Name:  inst.InstanceName,
            Type:  dsType,
            URL:   dsURL,
            OrgID: tenant.GrafanaOrgID,
        }
        
        if err := s.grafanaClient.CreateDatasource(ctx, datasource); err != nil {
            log.Error("创建 Grafana 数据源失败", "instance", inst.InstanceName, "error", err)
        }
    }
    
    return nil
}
```

## 4. 同步触发机制

### 4.1 自动同步

| 事件 | 触发时机 | 同步内容 |
|------|---------|---------|
| 创建租户 | 租户创建成功后 | VMuser, N9E Team, Grafana Org |
| 创建用户 | 用户创建成功后 | N9E User, Grafana OrgUser |
| 更新租户 | 租户信息变更 | VMuser 配额 |
| 删除租户 | 租户删除前 | 删除 N9E Team, Grafana Org |
| 删除用户 | 用户删除前 | 删除 N9E User, Grafana OrgUser |
| 创建实例 | 实例创建成功后 | N9E/Grafana 数据源 |
| 删除实例 | 实例删除前 | 删除 N9E/Grafana 数据源 |

```go
    }
    
    return nil
}
```

### 5.2 认证实现

```go
// 平台 JWT 认证
type JWTAuth struct {
    secret     string
    expiration time.Duration
}

func (j *JWTAuth) GenerateToken(user *User) (string, error) {
    claims := jwt.MapClaims{
        "user_id":   user.ID.String(),
        "username": user.Username,
        "tenant_id": user.TenantID,
        "role":     user.Role,
        "exp":      time.Now().Add(j.expiration).Unix(),
    }
    
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString([]byte(j.secret))
}

func (j *JWTAuth) ValidateToken(tokenString string) (*jwt.MapClaims, error) {
    token, err := jwt.Parse(tokenString, func(token *jwt.Token) (interface{}, error) {
        return []byte(j.secret), nil
    })
    
    if claims, ok := token.Claims.(jwt.MapClaims); ok && token.Valid {
        return &claims, nil
    }
    
    return nil, err
}
```

## 6. 同步失败处理

### 6.1 重试机制

```go
type SyncRetryConfig struct {
    MaxRetries int           `default:"3"`
    Interval   time.Duration `default:"5s"`
    Backoff    time.Duration `default:"10s"`
}

func (s *SyncService) syncWithRetry(ctx context.Context, fn func() error) error {
    var lastErr error
    
    for i := 0; i < s.retryConfig.MaxRetries; i++ {
        if err := fn(); err != nil {
            lastErr = err
            log.Warn("同步失败，准备重试", "retry", i+1, "error", err)
            time.Sleep(s.retryConfig.Interval + time.Duration(i)*s.retryConfig.Backoff)
            continue
        }
        return nil
    }
    
    // 记录最终失败
    s.recordSyncFailure(ctx, lastErr)
    return lastErr
}
```

```go
// 同步失败告警
func (s *SyncService) recordSyncFailure(ctx context.Context, err error) {
    // 记录到数据库
    record := &SyncRecord{
        TargetSystem: "n9e", // 或 grafana, vmuser
        Status:       "failed",
        ErrorMessage: err.Error(),
    }
    s.syncRepo.Create(ctx, record)
    
    // 发送告警通知
    alert := &Alert{
        RuleName:    "同步失败告警",
        Level:       "warning",
        Message:     fmt.Sprintf("用户同步失败: %s", err.Error()),
        Channel:     "dingtalk",
    }
    s.alertService.Send(ctx, alert)
}
```
