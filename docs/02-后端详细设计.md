```go
type TenantService interface {
    // 创建租户
    Create(ctx context.Context, req *CreateTenantRequest) (*Tenant, error)
    
    // 获取租户
    GetByID(ctx context.Context, id string) (*Tenant, error)
    GetByVMUserID(ctx context.Context, vmuserID string) (*Tenant, error)
    List(ctx context.Context, filter *TenantFilter) ([]*Tenant, error)
    
    // 更新租户
    Update(ctx context.Context, id string, req *UpdateTenantRequest) (*Tenant, error)
    
    // 删除租户
    Delete(ctx context.Context, id string) error
    
    // 同步到外部系统
    SyncToN9E(ctx context.Context, tenantID string) error
    SyncToGrafana(ctx context.Context, tenantID string) error
    SyncAll(ctx context.Context, tenantID string) error
    
    // 资源配额
    GetQuota(ctx context.Context, tenantID string) (*TenantQuota, error)
    UpdateQuota(ctx context.Context, tenantID string, quota *TenantQuota) error
}

// 创建租户核心流程
func (s *TenantServiceImpl) Create(ctx context.Context, req *CreateTenantRequest) (*Tenant, error) {
    // 1. 验证部门存在
    dept, err := s.deptRepo.GetByID(ctx, req.DeptID)
    if err != nil {
        return nil, errors.New("部门不存在")
    }
    
    // 2. 生成 VMuser ID
    vmuserID := generateVMUserID()
    vmuserKey := generateAPIKey()
    
    // 3. 创建租户记录
    tenant := &Tenant{
        TenantName: req.TenantName,
        DeptID:     req.DeptID,
        VMUserID:   vmuserID,
        VMUserKey:  vmuserKey,
        TemplateType: req.TemplateType,
        QuotaConfig: req.QuotaConfig,
        Status:     "creating",
    }
    if err := s.tenantRepo.Create(ctx, tenant); err != nil {
        return nil, err
    }
    
    // 4. 创建 K8s Namespace
    if err := s.k8sClient.CreateNamespace(ctx, tenant.Namespace()); err != nil {
        return nil, err
    }
    
    // 5. 配置 ResourceQuota
    if err := s.k8sClient.ApplyResourceQuota(ctx, tenant.Namespace(), tenant.Quota()); err != nil {
        return nil, err
    }
    
    // 6. 部署监控实例 (根据模板类型)
    if err := s.orchestrator.DeployTenantInstances(ctx, tenant); err != nil {
        return nil, err
    }
    
    // 7. 同步到 N9E
    if err := s.n9eClient.CreateTeam(ctx, vmuserID, tenant.TenantName); err != nil {
        // 日志记录，不阻塞
        log.Error("同步 N9E 失败", err)
    }
    
    // 8. 同步到 Grafana
    if err := s.grafanaClient.CreateOrg(ctx, vmuserID, tenant.TenantName); err != nil {
        log.Error("同步 Grafana 失败", err)
    }
    
    // 9. 更新租户状态
    tenant.Status = "active"
    s.tenantRepo.Update(ctx, tenant)
    
    return tenant, nil
}
```

```go
    // 共享版不需要部署新的 VM 资源
    // 只需要配置 vmauth 用户和配额

        return err
    }
    
    // 2. 配置 tenant 配额
    if err := o.vmOperator.UpdateTenantQuota(ctx, tenant.VMUserID, tenant.QuotaConfig); err != nil {
        return err
    }

    if err := o.helmClient.InstallRelease(ctx, "vl-"+tenant.VMUserID, "vm/victoria-logs-single", tenant.Namespace(), o.getVLValues(tenant)); err != nil {
        return err
    }
```

### 4.3 用户同步服务 (Sync Service)

```go
type SyncService interface {
    // 同步租户到 N9E
    SyncTenantToN9E(ctx context.Context, tenantID string) error
    
    // 同步租户到 Grafana
    SyncTenantToGrafana(ctx context.Context, tenantID string) error
    
    // 同步用户到 N9E
    SyncUserToN9E(ctx context.Context, userID string) error
    
    // 同步用户到 Grafana
    SyncUserToGrafana(ctx context.Context, userID string) error
    
    // 同步告警数据源
    SyncDatasourceToN9E(ctx context.Context, tenantID string) error
}

type SyncServiceImpl struct {
    n9eClient    N9EClient
    grafanaClient GrafanaClient
    tenantRepo   TenantRepository
    userRepo     UserRepository
}

// 同步租户到 N9E
func (s *SyncServiceImpl) SyncTenantToN9E(ctx context.Context, tenantID string) error {
    tenant, err := s.tenantRepo.GetByID(ctx, tenantID)
    if err != nil {
        return err
    }
    
    // 1. 创建 N9E 团队
    team := &N9ETeam{
        Name: tenant.VMUserID,
        Note: tenant.TenantName,
    }
    if err := s.n9eClient.CreateTeam(ctx, team); err != nil {
        return err
    }
    
    // 2. 创建 N9E 用户 (租户管理员)
    users, err := s.userRepo.ListByTenant(ctx, tenantID)
    for _, user := range users {
        n9eUser := &N9EUser{
            Ident:    user.ID.String(),
            Nick:     user.Username,
            Password: user.PasswordHash,
            Ntype:    0, // 用户
            TeamID:   team.ID,
        }
        s.n9eClient.CreateUser(ctx, n9eUser)
    }
    
    // 3. 同步数据源
    return s.SyncDatasourceToN9E(ctx, tenantID)
}

// 同步数据源到 N9E
func (s *SyncServiceImpl) SyncDatasourceToN9E(ctx context.Context, tenantID string) error {
    tenant, err := s.tenantRepo.GetByID(ctx, tenantID)
    if err != nil {
        return err
    }
    
    // 获取实例信息
    instances, err := s.instanceRepo.ListByTenant(ctx, tenantID)
    
    for _, inst := range instances {
        datasource := &N9EDatasource{
            Name:   inst.Name,
            Type:   inst.Type, // prometheus / loki
            URL:    inst.URL,
            Tenant: tenant.VMUserID,
        }
        if err := s.n9eClient.CreateDatasource(ctx, tenant.VMUserID, datasource); err != nil {
            log.Error("创建 N9E 数据源失败", err)
        }
    }
    
    return nil
}

// 同步租户到 Grafana
func (s *SyncServiceImpl) SyncTenantToGrafana(ctx context.Context, tenantID string) error {
    tenant, err := s.tenantRepo.GetByID(ctx, tenantID)
    if err != nil {
        return err
    }
    
    // 1. 创建 Grafana 组织
    org := &GrafanaOrg{
        Name: tenant.VMUserID,
        NameLabel: tenant.TenantName,
    }
    if err := s.grafanaClient.CreateOrg(ctx, org); err != nil {
        return err
    }
    
    // 2. 添加用户到组织
    users, err := s.userRepo.ListByTenant(ctx, tenantID)
    for _, user := range users {
        orgUser := &GrafanaOrgUser{
            Login:  user.ID.String(),
            Role:   string(user.Role), // admin/editor/viewer
        }
        if user.Role == "admin" {
            s.grafanaClient.SetOrgAdmin(ctx, tenant.VMUserID, user.ID.String())
        } else {
            s.grafanaClient.AddOrgUser(ctx, tenant.VMUserID, orgUser)
        }
    }
    
    // 3. 添加数据源
    instances, _ := s.instanceRepo.ListByTenant(ctx, tenantID)
    for _, inst := range instances {
        if inst.Type == "metrics" {
            ds := &GrafanaDatasource{
                Name:  inst.Name,
                Type:  "prometheus",
                URL:   inst.URL,
            }
            s.grafanaClient.CreateDatasource(ctx, tenant.VMUserID, ds)
        } else if inst.Type == "logs" {
            ds := &GrafanaDatasource{
                Name:  inst.Name,
                Type:  "loki",
                URL:   inst.URL,
            }
            s.grafanaClient.CreateDatasource(ctx, tenant.VMUserID, ds)
        }
    }
    
    return nil
}
```

```go
type VMUserManager interface {
    // 创建 VMuser
    CreateVMUser(ctx context.Context, tenant *model.Tenant) error
    
    // 更新 VMuser 配额
    UpdateVMUserQuota(ctx context.Context, vmuserID string, quota *VMQuota) error
    
    // 删除 VMuser
    DeleteVMUser(ctx context.Context, vmuserID string) error
    
    // 获取 vmauth 访问地址
    GetAccessEndpoint(vmuserID string) string
}

type VMUserManagerImpl struct {
    vmOperator VMOperatorClient
}

// 创建 VMuser
func (m *VMUserManagerImpl) CreateVMUser(ctx context.Context, tenant *model.Tenant) error {
    // VMuser 通过 vmauth 配置管理
    // 使用 VM Operator 的 VMAuth CRD
    
    vmauthConfig := &VMAuthConfig{
        // 配置 users 列表
        Users: []VMAuthUser{
            {
                Name:       tenant.VMUserID,
                Password:   tenant.VMUserKey,
                URLPrefixes: []string{
                    "/insert/" + tenant.VMUserID,
                    "/select/" + tenant.VMUserID,
                },
            },
        },
    }
    
    return m.vmOperator.UpdateVMAuth(ctx, "platform", vmauthConfig)
}

// 获取访问地址
func (m *VMUserManagerImpl) GetAccessEndpoint(vmuserID string) string {
    return fmt.Sprintf("http://vmauth.platform:8421/insert/%s", vmuserID)
}
```

```sql
    tenant_name     VARCHAR(255) NOT NULL,
    dept_id         UUID UNIQUE REFERENCES ops_departments(id),
    vmuser_id       VARCHAR(100) UNIQUE NOT NULL,
    vmuser_key      VARCHAR(255) NOT NULL,
    template_type   VARCHAR(50) NOT NULL,
    quota_config    JSONB,
    status          VARCHAR(20) DEFAULT 'creating',
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);

-- 实例表
CREATE TABLE ops_instances (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES ops_tenants(id),
    instance_name   VARCHAR(255) NOT NULL,
    instance_type  VARCHAR(50) NOT NULL, -- metrics, logs, visual, alert
    template_type   VARCHAR(50) NOT NULL, -- shared, dedicated_single, dedicated_cluster
    release_name    VARCHAR(100) NOT NULL, -- Helm release name

    status          VARCHAR(20) DEFAULT 'creating',
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()

-- 告警规则表（预留，告警由 N9E 独立管理）
CREATE TABLE alert_rules (

    updated_at      TIMESTAMP DEFAULT NOW()
);

-- 通知渠道表（预留，告警由 N9E 独立管理）
CREATE TABLE notification_channels (
    id              UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    tenant_id       UUID NOT NULL REFERENCES ops_tenants(id),
    channel_name    VARCHAR(255) NOT NULL,
    channel_type    VARCHAR(50) NOT NULL, -- dingtalk, email, slack, sms
    config          JSONB NOT NULL,
    created_at      TIMESTAMP DEFAULT NOW(),
    updated_at      TIMESTAMP DEFAULT NOW()
);
```

## 7. 部署架构

### 7.1 平台初始化

平台部署顺序:

1. **基础设施**: PostgreSQL, Redis, etcd
2. **全局 VM**: 全局 VMCluster + vmauth
3. **N9E**: N9E Server
4. **API 服务**: 后端 API 服务
5. **前端**: Web 控制台

### 7.2 租户部署

租户创建时部署顺序:

```
1. 创建 Namespace
2. 配置 ResourceQuota
3. 部署 VM 资源 (根据模板类型)
4. 部署 VictoriaLogs
5. 部署 Grafana
6. 部署 n9e-edge
7. （可选）配置 N9E 告警数据源
8. 同步到 Grafana
```

### 7.3 扩缩容

| 操作 | 实现方式 |
|------|---------|
| 垂直扩容 | 更新 Helm values，helm upgrade |
| 水平扩容 | 修改 replicaCount，kubectl scale |
| 存储扩容 | 扩大 PVC 大小 |
| 模板升级 | 创建新实例，迁移数据，删除旧实例 |
